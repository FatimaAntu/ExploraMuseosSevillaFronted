import { Component, OnInit } from '@angular/core';
import { ContactoService, MensajeContacto } from '../../services/contacto.service';
import { CommonModule } from '@angular/common';

/**
 * Componente para que el administrador visualice los mensajes enviados a través
 * del formulario de contacto. Muestra los mensajes en orden descendente según el ID,
 * es decir, los más recientes primero.
 */
@Component({
  selector: 'app-mensajes-admin',
  templateUrl: './mensajes-admin.component.html',
  styleUrls: ['./mensajes-admin.component.css'],
  standalone: true,
  imports: [CommonModule]
})
export class MensajesAdminComponent implements OnInit {
  /**
   * Array que contiene los mensajes recibidos desde el backend.
   */
  mensajes: MensajeContacto[] = [];

  /**
   * Indicador para mostrar estado de carga mientras se obtienen los mensajes.
   */
  cargando = true;

  /**
   * Constructor con inyección del servicio de contacto para obtener los mensajes.
   * @param contactoService Servicio para gestión de mensajes de contacto
   */
  constructor(private contactoService: ContactoService) {}

  /**
   * Ciclo de vida OnInit que lanza la llamada al servicio para cargar los mensajes.
   * Ordena los mensajes para mostrar primero los más recientes.
   * En caso de error muestra un mensaje en consola y termina la carga.
   */
  ngOnInit(): void {
    console.log('MensajesAdminComponent cargando...');
    this.contactoService.obtenerMensajes().subscribe({
      next: (data) => {
        this.mensajes = data.sort((a, b) => b.id - a.id);
        this.cargando = false;
      },
      error: (error) => {
        console.error('Error al obtener mensajes', error);
        this.cargando = false;
      }
    });
  }
}

<div class="p-6 bg-gray-900 text-white rounded-lg shadow-lg max-w-4xl mx-auto">
  <h2>Mensajes de Contacto</h2>

  <div *ngIf="cargando" class="texto-estado">Cargando mensajes...</div>

  <div *ngIf="!cargando && mensajes.length === 0" class="texto-estado">
    No hay mensajes por el momento.
  </div>

  <div *ngFor="let msg of mensajes" class="mensaje">
    <div class="mensaje-header">
      <h3 class="mensaje-nombre">{{ msg.nombre }}</h3>
      <span class="mensaje-email">{{ msg.email }}</span>
    </div>
    <p class="mensaje-texto">{{ msg.mensaje }}</p>
  </div>
</div>
